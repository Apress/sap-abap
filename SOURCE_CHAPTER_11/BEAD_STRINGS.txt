method BEAD_STRINGS.

* CLASS -  YCL_CH11_CL_STRING_OPERATIONS
* METHOD - BEAD_STRINGS
*    IMPORTING
*      by reference STR_TAB TYPE STRING_TABLE
*      by value SEPARATOR TYPE C default ' '
*    RETURNING
*      by value STRNG TYPE STRING
*    EXCEPTIONS
*      NO_DATA .


 DESCRIBE TABLE STR_TAB..

 IF SY-TFILL = 0.
  MESSAGE S035(YCL_CH07_MCLASS01) RAISING
   NO_DATA.
 ENDIF.

 LOOP AT STR_TAB INTO STR.
   IF SY-TABIX > 1.
     CONCATENATE STRNG STR INTO STRNG SEPARATED
      BY SEPARATOR.
   ELSE.
     STRNG = STR.
   ENDIF.
 ENDLOOP.

endmethod.
